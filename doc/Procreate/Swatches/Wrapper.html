def colors(format: nil)
  format = format.to_sym unless format.nil?
  format = nil unless format.in?(AVAILABLE_COLOR_FORMATS)

  return @colors if format.nil?

  @colors.map do |color|
    color.send("to_#{format}")
  end
end
